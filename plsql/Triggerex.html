CREATE OR REPLACE TRIGGER cstud
BEFORE UPDATE ON student
fOR EACH ROW
BEGIN
IF :NEW.Marks > 300 THEN
:NEW.Marks := :OLD.Marks;
dbms_output.put_line('Marks should be less than 300!');
END IF;
END;
